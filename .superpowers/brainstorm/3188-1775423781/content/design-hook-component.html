useExport — API do hook

src/features/export/useExport.ts
interface UseExportOptions {
  filename: string           // ex: "piloto-testes-cover" (sem .png)
}

interface UseExportReturn {
  exportOne: () => Promise<void>   // exporta o slide do ref
  exportAll: () => Promise<void>   // exporta todos os refs em sequência
  isExporting: boolean             // true durante qualquer export em andamento
}

function useExport(
  ref: RefObject<HTMLElement>,        // ref do slide atual
  allRefs: RefObject<HTMLElement>[],  // todos os refs (para exportAll)
  allFilenames: string[],             // filenames paralelos a allRefs
  options: UseExportOptions
): UseExportReturn

O hook usa toPng(ref.current, { pixelRatio: 2 }) do html-to-image. O download é feito via anchor element programático. isExporting bloqueia novos cliques enquanto um export está em andamento.
